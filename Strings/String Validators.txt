Problem Name: String Validators

Link: https://www.hackerrank.com/challenges/string-validators/problem

Solution:

if __name__ == '__main__':
    s = input()
    isalnum = isalpha = isdigit = islower = isupper = False
    for i in range(len(s)):
       if s[i].isalnum() == True:
           isalnum = True
       if s[i].isalpha() == True:
           isalpha = True
       if s[i].isdigit() == True:
            isdigit = True
       if s[i].islower() == True:
            islower = True
       if s[i].isupper() == True:
        isupper = True

    print(isalnum)
    print(isalpha)
    print(isdigit)
    print(islower)
    print(isupper)
